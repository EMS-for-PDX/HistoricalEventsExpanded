add_small_peril_effect = {
    if = {
        limit = {
			any_owned_story = {
				story_type = story_anarchy_at_samarra
			}
        }
		if = {
			limit = {	
				NOT = { has_variable = anarchy_peril_count }
			}
			set_variable = {
				name = anarchy_peril_count
				value = 1 
			}
		}
		else = {
			change_variable = {
				name = anarchy_peril_count
				add = 1
			}
		}
    }
    custom_tooltip = anarchy_peril_explanation_tt
}
