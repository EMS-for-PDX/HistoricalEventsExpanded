hee_great_summer_army_scripted_effect = {
	character:168650 = { save_scope_as = guthrum }
	
	scope:guthrum = {
		#Give the adventurer a dynamic title to tide them over.
		#get_title = d_aghlabid_raiders
		
		create_dynamic_title = {
			tier = duchy
			name = { SCANDANAVIAN_ADVENTURER_NEUTRAL_NAME }
		}
	}
	
	title:d_great_summer_army = { save_scope_as = new_title }
	
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change
		add_claim_on_loss = no
	}
	
	scope:new_title = {
		change_title_holder = {
			holder = scope:guthrum
			change = scope:change
		}
	}
	
	resolve_title_and_vassal_change = scope:change
	
	scope:new_title = {
		generate_coa = yes
		set_variable = {
			name = temporary_title
			value = yes
		}
	}
	
	title:d_normandy.holder.liege = {
		save_scope_as = normandy_target_holder
	}
	
	#Declare the war.
	scope:guthrum = {
		start_war = {
			casus_belli = ducal_conquest_cb
			target = scope:normandy_target_holder
			target_title = title:d_normandy
		}
	}
	
	#Setup event troops.
	scope:guthrum = {
		spawn_army = {
			uses_supply = no
			inheritable = no
			name = great_summer_army_event_troops
			levies = {
				value = 2950
			}
			men_at_arms = {
				type = light_footmen
				stacks = 5
			}
			men_at_arms = {
				type = light_footmen
				stacks = 5
			}
			men_at_arms = {
				type = bowmen
				stacks = 5
			}
			men_at_arms = {
				type = bowmen
				stacks = 5
			}
			men_at_arms = {
				type = huscarl
				stacks = 5
			}
			men_at_arms = {
				type = huscarl
				stacks = 5
			}
			men_at_arms = {
				type = huscarl
				stacks = 5
			}
			men_at_arms = {
				type = huscarl
				stacks = 5
			}
			men_at_arms = {
				type = mangonel
				stacks = 5
			}
			location = province:688 #The Wash, off the coast of East Anglia
		}
		# Suppress startup error false-positive
		if = {
			limit = { has_variable = random_location }
			# Do nothing.
		}
		# Loan a little gold in case they're in debt/to keep them going.
 		scope:rollo = { add_gold = 550 }
	}
	
}
