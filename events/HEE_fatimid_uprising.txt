namespace = HEE_fatimid_uprising

### Spawn Fatimids
HEE_fatimid_uprising.0001 = {
	scope = none
	hidden = yes

	immediate = {
		debug_log = "Spawning the Fatimids before 1066 start date"
		debug_log_date = yes
		spawn_fatimids_character_effect = yes
		global_var:ismaili_missionary = {
			trigger_event = {
				id = HEE_fatimid_uprising.0112
			}
		}
	}
}

HEE_fatimid_uprising.0112 = {
	scope = none
	hidden = yes

	immediate = {
		every_county_in_region = {
			region = core_qarmatian_region
			limit = { faith = faith:ismaili }
			random = {
				chance = 80
				set_county_faith = faith:qarmatian
			}
		}
		global_var:qarmatian_founder = { 
			#give_ghaznavid_land_effect = yes
			add_pressed_claim = title:d_bahrain
			add_pressed_claim = title:d_al-hasa
			trigger_event = {
				id = HEE_fatimid_uprising.0011 #trigger the actual war stuff
			}
		}
	}
}

### Spawn Title 
HEE_fatimid_uprising.0011 = {
	scope = none
	hidden = yes

	immediate = {
		global_var:qarmatian_founder = {
			create_dynamic_title = {
				tier = kingdom
				name = "Qarmatians"
			}
			create_title_and_vassal_change = {
				type = created
				save_scope_as = change
				add_claim_on_loss = no
			}
			scope:new_title = {
				set_capital_county = title:c_al-hasa
				set_landless_title = yes
				set_no_automatic_claims = yes
				set_definitive_form = yes
				set_can_be_named_after_dynasty = no

				change_title_holder = {
					holder = global_var:qarmatian_founder
					change = scope:change
				}
			}
		
			resolve_title_and_vassal_change = scope:change
		
			scope:new_title = { 
				set_coa = global_var:qarmatian_founder.house
				set_color_from_title = title:d_lobi
			}
			
			add_realm_law_skip_effects = high_partition_succession_law
		
			set_global_variable = {
				name = qarmatian_realm_title
				value = scope:new_title
			}
			save_scope_as = story_owner # Do I really need to do this? Isn't it set automatically?
			create_story = story_qarmatian_invasion
		}
	}
}
