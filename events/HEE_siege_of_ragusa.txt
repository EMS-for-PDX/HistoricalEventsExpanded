HEE_siege_of_ragusa.0001 = { #Ragusa sends offer to Byz Emperor
	window = big_event_window
	type = character_event
	title = HEE_siege_of_ragusa.0001.t
	desc = HEE_siege_of_ragusa.0001.desc
	theme = diplomacy # The event centers around considering Diplomatic solutions to the Siege
	override_background = { reference = corridor_day }
	left_portrait = {
		character = root
		animation = stress
	}


	trigger = {
		is_alive = yes
		is_landed = yes
		is_independent_ruler = yes
		is_at_war = yes
		is_imprisoned = no
	}

	#Resend
	on_trigger_fail = {
		if = {
			trigger_event = {
				id = HEE_siege_of_ragusa.0001
				days = 7 #~1 week
			}
		}
	}

	immediate = {
		root = { save_scope_as = ragusa_leader }
		title:e_byzantium.holder = { save_scope_as = roman_emperor }
		title:d_aghlabid_raiders.holder = { save_scope_as = aghlabid_raiders }
	}

	option = {
		name = HEE_siege_of_ragusa.0001.a
		
		custom_tooltip = reach_out_to_roman_emperor_for_aid

		scope:ragusa_leader = {
			random_courtier_or_guest = {
				limit = {
					is_available = yes
				}
				save_scope_as = companion1
			}
			random_courtier_or_guest = {
				limit = {
					this != scope:companion1
					is_available = yes
				}
				save_scope_as = companion2
			}
			random_courtier_or_guest = {
				limit = {
					NOR = {
						this = scope:companion1
						this = scope:companion2
					}
					is_available = yes
				}
				save_scope_as = companion3
			}
			start_travel_plan = {
				destination = scope:roman_emperor.capital_province
				companion = scope:companion1
				companion = scope:companion2
				companion = scope:companion3
				on_arrival_event = HEE_siege_of_ragusa.0002
				on_arrival_destinations = all_but_last
				return_trip = yes # One way
			}
		}
		
		stress_impact = {
			stubborn = medium_stress_impact_gain
			paranoid = medium_stress_impact_gain
			arrogant = medium_stress_impact_gain
			brave = minor_stress_impact_loss
			diplomat = minor_stress_impact_loss
			humble = minor_stress_impact_loss
		}
		
		ai_chance = {
			base = 1000
		}
	}

	option = {
		name = HEE_siege_of_ragusa.0001.b
		
		custom_tooltip = attempt_to_fend_off_aghlabids_without_aid

		stress_impact = {

		}
		
		ai_chance = {
			base = 0
		}
	}
}
